To install OpenBabel
====================
sudo wget http://sourceforge.net/projects/openbabel/files/openbabel/2.3.1/openbabel-2.3.1.tar.gz
sudo tar zxvf openbabel-2.3.1.tar.gz

sudo mv openbabel-2.3.1 ob-src
sudo mkdir ob-build

cd ob-build

sudo apt-get install cmake
sudo apt-get install make
sudo cmake ../ob-src 2>&1 | sudo tee cmake.out
sudo make 2>&1 | sudo tee make.out
sudo make install

export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/usr/local/lib
